function peak(arr) {
    for (let i = 1; i < arr.length - 1; i++) {
        let leftSum = arr.slice(0, i).reduce((a, b) => a + b, 0);
        let rightSum = arr.slice(i + 1).reduce((a, b) => a + b, 0);
        if (leftSum === rightSum) return i;
    }
    return -1;
}
